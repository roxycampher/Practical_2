Question 2

library(MASS)
gal <- galaxies

B <- 1000
n <- length(gal)

# generate bootstrap medians
m <- function(n, gal) {
  sam <- sample(gal, n, replace = TRUE)
  median(sam)
}

c2 <- makeCluster(3)
registerDoParallel(c2)

system.time({
b_m <- foreach(i = 1:B, .combine = rbind) %dopar% {
  m(n, gal)
}
})
   user  system elapsed 
   0.22    0.04    0.31 
stopCluster(c2)

tboot <- numeric(B)

system.time({
  for (i in 1:B) {
  tboot[i] <- median(sample(gal, length(gal), replace = TRUE))
}
})
   user  system elapsed 
   0.05    0.00    0.05

In this instance serial processing is much faster as Bootstrapping is relatively small. So paralellization is not always more beneficial due to the complex overhead involved in splitting and executing tasks between threads.

Question 3

set.seed(1)

R <- 1000
B <- 1000

cl <- makeCluster(3)
registerDoParallel(cl)

hits <- foreach(r = 1:R, .combine = "+") %dopar% {
  x <- rexp(50, rate = 1)

  boot_means <- replicate(B, mean(sample(x, 50, replace = TRUE)))
  CI <- quantile(boot_means, c(0.025, 0.975), names = FALSE)

  as.integer(CI[1] <= 1 && 1 <= CI[2])
}

stopCluster(cl)

coverage <- hits / R
coverage
[1] 0.93

Question 5

library(parallel)

set.seed(1234)

run_replicate <- function() {
  # serial base R
  replicate(3, max(rnorm(5)))
}

run_foreach <- function() {
  it <- irnorm(5)
  foreach(i = icount(3), .combine = c) %do% {
    max(nextElem(it))
  }
}

run_parLapply <- function(cl) {
  unlist(parLapply(cl, 1:3, function(i) max(rnorm(5))))
}

run_foreach_par <- function() {
  foreach(i = 1:3, .combine = c) %dopar% max(rnorm(5))
}

R <- 5000

# 1) replicate
t_rep <- system.time({
  for (r in 1:R) run_replicate()
})

# 2) foreach
t_for_do <- system.time({
  for (r in 1:R) run_foreach()
})

cl <- makeCluster(3)
registerDoParallel(cl)

# 3) parLapply
t_parL <- system.time({
  for (r in 1:R) run_parLapply(cl)
})

# 4) foreach
t_for_par <- system.time({
  for (r in 1:R) run_foreach_par()
})

stopCluster(cl)

times <- rbind(
  replicate = t_rep[1:3],
  foreach_do = t_for_do[1:3],
  parLapply = t_parL[1:3],
  foreach_dopar = t_for_par[1:3]
)
times
              user.self sys.self elapsed
replicate          0.09     0.03    0.12
foreach_do        10.34     0.14   10.56
parLapply          3.26     0.59    5.19
foreach_dopar     17.79     0.59   21.48

Here we see that replicate ran the fastest since it has very low overhead and no inter-process communication.

foreach_do is sequential but slower than replicate as it has extra overhead which makes the run time slower.

parLapply is faster than foreach_do but still slower than sequential computation due to overhead associated with creating clusters and inter-process communication.

foreach_dopar is the slowest since it has parallel costs and framework overhead.

This demonstrates that parallel computing is only beneficial when the computational workload per task is sufficiently large to offset parallel overhead costs.
